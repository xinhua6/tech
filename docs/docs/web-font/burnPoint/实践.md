// 获取当前操作系统
function getOS() { 
    var os;
    if (navigator.userAgent.indexOf('Android') > -1 || navigator.userAgent.indexOf('Linux') > -1) {
        os = 'Android';
    } else if (navigator.userAgent.indexOf('iPhone') > -1||navigator.userAgent.indexOf('iPad') > -1) {
        os = 'iOS';
    } else if (navigator.userAgent.indexOf('Windows Phone') > -1) {
        os = 'WP';
    } else {
        os = 'Others';
    }
    return os;
}

tips：有些场景无法获取上一页的信息  
1. 直接在浏览器地址栏中输入地址；
2. 使用location.reload()刷新（location.href或者location.replace()刷新有信息）；
3. 在微信对话框中，点击链接进入微信自身的浏览器；
4. 扫码进入QQ或者微信的浏览器；
5. 直接新窗口打开一个页面； 2017.8.3更新 新版本Chrome测试，新窗口页面依然有document.referrer
6. 从https的网站直接进入一个http协议的网站（Chrome下亲测）；
7. a标签设置rel="noreferrer"（兼容IE7+）；
8. meta标签来控制不让浏览器发送referer；

            function getPluginName() {
                var info = "";
                var plugins = navigator.plugins;
                if (plugins.length > 0) {
                    for (i = 0; i < navigator.plugins.length; i++) {
                        info += navigator.plugins[i].name + ";";
                    }
                }
                return info;
            }
            //检查是否安装了某插件，如果安装了返回版本号
            function checkePlugs(pluginname) {
                var f = "-"
                var plugins = navigator.plugins;
                if (plugins.length > 0) {
                    for (i = 0; i < navigator.plugins.length; i++) {
                        if (navigator.plugins[i].name.indexOf(pluginname) >= 0) {
                            f = navigator.plugins[i].description.split(pluginname)[1];
                            return f;
                            break;
                        }
                    }
                }
                return false;
            }

            function raminfo() {

                        speed: memory.item().Speed

                info += "<tr style='color: red'><td >可用物理内存：" + physicMenCap + "M</td></tr>";
